package hospital.management.system;

import javax.swing.*;
import java.awt.*;
import java.sql.*;
import net.proteanit.sql.DbUtils;

public class InventoryManagement extends JFrame {
    private final JTable table;
    private JButton refreshBtn, backBtn;

    public InventoryManagement() {
        setSize(800, 600);
        setLocation(400, 100);

        JPanel panel = new JPanel();
        panel.setBounds(0, 0, 800, 600);
        panel.setBackground(new Color(90, 156, 163));
        panel.setLayout(null);
        add(panel);

        // Table
        table = new JTable();
        JScrollPane scrollPane = new JScrollPane(table);
        scrollPane.setBounds(20, 20, 750, 400);
        panel.add(scrollPane);

        // Buttons
        refreshBtn = new JButton("Refresh");
        refreshBtn.setBounds(200, 450, 120, 30);
        refreshBtn.addActionListener(e -> loadInventory());
        panel.add(refreshBtn);

        backBtn = new JButton("Back");
        backBtn.setBounds(400, 450, 120, 30);
        backBtn.addActionListener(e -> setVisible(false));
        panel.add(backBtn);

        loadInventory();
        setVisible(true);
    }

    private void loadInventory() {
        try {
            conn c = new conn();
            ResultSet rs = c.s.executeQuery("SELECT * FROM inventory");
            table.setModel(DbUtils.resultSetToTableModel(rs));
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}
